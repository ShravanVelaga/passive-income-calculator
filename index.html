        function calculateTax(income) {

            if (income <= 1200000) return 0; // Budget 2025 Rebate

            let tax = 0;

            if (income > 400000) tax += (Math.min(income, 800000) - 400000) * 0.05;

            if (income > 800000) tax += (Math.min(income, 1200000) - 800000) * 0.10;

            if (income > 1200000) tax += (Math.min(income, 1600000) - 1200000) * 0.15;

            if (income > 1600000) tax += (Math.min(income, 2000000) - 1600000) * 0.20;

            if (income > 2000000) tax += (income - 2000000) * 0.25;

            return tax;

        }


        function calculate() {

            const p = parseFloat(document.getElementById('principal').value) || 0;

            const other = parseFloat(document.getElementById('otherInc').value) || 0;


            // FD 9%

            const fdGross = p * 0.09;

            const totalWithFD = other + fdGross;

            const totalTaxWithFD = calculateTax(totalWithFD);

            const taxOnlyOther = calculateTax(other);

            const fdTax = totalTaxWithFD - taxOnlyOther;

            const fdNetMo = (fdGross - fdTax) / 12;


            // Arbitrage 7.5%

            const arbGross = p * 0.075;

            const arbTax = arbGross > 125000 ? (arbGross - 125000) * 0.125 : 0;

            const arbNetMo = (arbGross - arbTax) / 12;


            // UI Update

            document.getElementById('fdGross').innerText = '₹' + Math.round(fdGross).toLocaleString('en-IN');

            document.getElementById('fdTax').innerText = '₹' + Math.round(fdTax).toLocaleString('en-IN');

            document.getElementById('fdNet').innerText = '₹' + Math.round(fdNetMo).toLocaleString('en-IN');

            

            document.getElementById('arbGross').innerText = '₹' + Math.round(arbGross).toLocaleString('en-IN');

            document.getElementById('arbTax').innerText = '₹' + Math.round(arbTax).toLocaleString('en-IN');

            document.getElementById('arbNet').innerText = '₹' + Math.round(arbNetMo).toLocaleString('en-IN');


            // Safety Alert

            const safety = document.getElementById('fdSafety');

            if (p > 500000) {

                safety.innerText = "UNINSURED";

                safety.className = "text-[10px] font-bold px-2 py-1 rounded bg-red-100 text-red-700";

            } else {

                safety.innerText = "RBI INSURED";

                safety.className = "text-[10px] font-bold px-2 py-1 rounded bg-green-100 text-green-700";

            }

        }


        function calculateGoal() {

            const target = parseFloat(document.getElementById('targetMonthly').value) || 0;

            if (target > 0) {

                const required = (target * 12) / 0.08; // Average 8% conservative return

                document.getElementById('goalResult').innerText = "To earn this, you need ~₹" + Math.round(required).toLocaleString('en-IN') + " at 8% yield.";

            }

        }

        calculate();
